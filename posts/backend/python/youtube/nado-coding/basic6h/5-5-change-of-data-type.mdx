# 자료구조의 변경

```python
menu = {"커피", "우유", "주스"}

print(menu)

# 출력값 : {'주스', '커피', '우유'}


# type 확인
print(menu, type(menu))

# 출력값 : {'커피', '주스', '우유'} <class 'set'>


# type : set → list
menu = list(menu)
print(menu, type(menu))

# 출력값 : ['커피', '우유', '주스'] <class 'list'>


# type : list → tuple
menu = tuple(menu)
print(menu, type(menu))

# 출력값 : ('우유', '커피', '주스') <class 'tuple'>


# type : tuple → set
menu = set(menu)
print(menu, type(menu))

# 출력값 : {'커피', '주스', '우유'} <class 'set'>
```
